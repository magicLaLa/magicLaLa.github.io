_renderDayRect(calendarModel: CalendarModel, rangeData: CalendarParsedDateRangeInfo, group: graphic.Group) {

        const rect = new graphic.Rect({
            shape: {

                y: point[1],
                width: sw,
                height: sh
            },
            cursor: 'default',
            style: itemRectStyleModel
        });
